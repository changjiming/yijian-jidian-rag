### 1. 🔍 增强检索准确率
**当前问题**：有时检索到的文档并非最相关的
**改进方案**：
- [ ] 增加BM25稀疏检索与向量检索混合
- [ ] 实现重排序（Re-ranking）
- [ ] 优化Embedding模型（考虑使用bge-large-zh）

**代码示例**：
```python
# 简单BM25实现
from rank_bm25 import BM25Okapi

def bm25_search(query, documents, k=3):
    tokenized_docs = [doc['page_content'].split() for doc in documents]
    bm25 = BM25Okapi(tokenized_docs)
    scores = bm25.get_scores(query.split())
    top_indices = scores.argsort()[-k:][::-1]
    return [documents[i] for i in top_indices]

# 混合检索结果
def hybrid_search(query, vector_store, k=3):
    # 向量检索
    vector_results = vector_store.similarity_search(query, k=k*2)
    # BM25检索
    bm25_results = bm25_search(query, vector_store.documents, k=k*2)
    # 合并结果，去重
    seen = set()
    final_results = []
    for doc in vector_results + bm25_results:
        content_hash = hash(doc['page_content'])
        if content_hash not in seen:
            seen.add(content_hash)
            final_results.append(doc)
        if len(final_results) &gt;= k:
            break
    return final_results
```

```python
import pdfplumber

def load_pdf_with_page_numbers(pdf_path):
    docs = []
    with pdfplumber.open(pdf_path) as pdf:
        for page_num, page in enumerate(pdf.pages, 1):
            text = page.extract_text()
            if text:
                docs.append({
                    'page_content': text,
                    'metadata': {
                        'page_number': page_num,
                        'source': pdf_path
                    }
                })
    return docs
```

```python
# 简单的对话历史管理
import json
from datetime import datetime

class ChatHistory:
    def __init__(self, save_path="chat_history.json"):
        self.save_path = save_path
        self.conversations = self._load()
    
    def _load(self):
        if Path(self.save_path).exists():
            with open(self.save_path, 'r', encoding='utf-8') as f:
                return json.load(f)
        return []
    
    def save_conversation(self, question, answer, context):
        self.conversations.append({
            'timestamp': datetime.now().isoformat(),
            'question': question,
            'answer': answer,
            'context': [doc['page_content'][:100] + "..." for doc in context]
        })
        with open(self.save_path, 'w', encoding='utf-8') as f:
            json.dump(self.conversations, f, ensure_ascii=False, indent=2)
```
